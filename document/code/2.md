```python
def solution(m, n, puddles):
    answer = [[0]*n for _ in range(m)]
    answer[0][0] = 1
    
    for i in range(1,m+n-1):
        minJ = max(0,i+1-n)
        maxJ = min(i+1,m)
        
        for j in range(minJ,maxJ):
            if [j+1,i-j+1] not in puddles:
                answer[j][i-j] = answer[j-1][i-j] + answer[j][i-j-1]
                
    return answer[m-1][n-1] % 1000000007
```
